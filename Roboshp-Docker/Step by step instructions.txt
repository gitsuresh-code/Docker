docker network ls # list of the networks available on docker
docker network create roboshop # to create own (roboshop) network
docker network disconnect bridge catalogue # to disconnect the container from bridge network (catalogue - container name)
docker network connect roboshop catalogue # to connect the catalogue container to roboshop network (own network which was created)

MongodB::
docker build -t mongodb:v1 .
docker run -d --network roboshop --name mongodb mongodb:v1 

Catalogue::
docker build -t catalogue:v1 . #to build the catalogue image
docker run -d --network roboshop --name catalogue catalogue:v1  #it will run the image to create catalogue container and start it
docker exec -it catalogue bash #to login into catalogue container
curl http://localhost:8080/health #to check the app running status
